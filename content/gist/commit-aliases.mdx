```shell title="conventional_commit_aliases.sh"
gcap() {
    git add . && git commit -m "$*" && git push
}
gnew() {
    gcap "feat: $@"
}
gimp() {
    gcap "improve: $@"
}
gfix() {
    gcap "fix: $@"
}
grlz() {
    gcap "release: $@"
}
gdoc() {
    gcap "docs: $@"
}
gtst() {
    gcap "test: $@"
}
gbrk() {
    gcap "BREAKING CHANGE: $@"
}
gchr() {
    gcap "chore: $@"
}
gtype() {
    NORMAL='\033[0;39m'
    GREEN='\033[0;32m'
    echo "$GREEN gnew$NORMAL — feat
    $GREEN gimp$NORMAL — improve
    $GREEN gfix$NORMAL — fix
    $GREEN grlz$NORMAL — release
    $GREEN gdoc$NORMAL — docs
    $GREEN gtst$NORMAL — test
    $GREEN gbrk$NORMAL — BREAKING CHANGE
    $GREEN gchr$NORMAL — chore"
}

```
