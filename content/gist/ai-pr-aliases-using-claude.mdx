# Spinner helper function for showing progress
spinner() {
  local pid=$1
  local delay=0.1
  local spinstr='|/-\'
  while kill -0 $pid 2>/dev/null; do
    local temp=${spinstr#?}
    printf " [%c]  " "$spinstr"
    local spinstr=$temp${spinstr%"$temp"}
    sleep $delay
    printf "\b\b\b\b\b\b"
  done
  printf "       \b\b\b\b\b\b"
}

aiprdevelop() {
  if ! git rev-parse --is-inside-work-tree &>/dev/null; then
    echo "Not inside a Git repository."
    return 1
  fi

  ahead_count=$(git rev-list --left-right --count develop...HEAD 2>/dev/null | awk '{print $2}')
  if [ "$ahead_count" -eq 0 ]; then
    echo "No commits ahead of the develop branch to create a PR."
    return 1
  fi

  system_prompt="system: you are to make a draft pull request against the develop branch. title should use conventional commit style, all smallcase. description should have no attribution. do not assign, label, or mention anyone. always create the pr as draft."

  echo "Generating draft pull request description via AI..."
  claude --verbose --dangerously-skip-permissions -p "$system_prompt" &
  spinner $!
  echo -e "\nDraft PR generation complete."

  exec zsh
}


aiprmain() {
  if ! git rev-parse --is-inside-work-tree &>/dev/null; then

  ahead_count=$(git rev-list --left-right --count main...HEAD 2>/dev/null | awk '{print $2}')

  fi

  system_prompt="system: you are to make a draft pull request against the main branch. title should use conventional commit style, all smallcase. description should have no attribution. do not assign, label, or mention anyone. always create the pr as draft."

}
