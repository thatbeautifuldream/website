```yaml title="amplify.yml"
version: 1
frontend:
  phases:
    preBuild:
      commands:
        - npx pnpm install
    build:
      commands:
        - npx pnpm build
        - rm -f node_modules/@swc/core-linux-x64-gnu/swc.linux-x64-gnu.node
        - rm -f node_modules/@swc/core-linux-x64-musl/swc.linux-x64-musl.node
  artifacts:
    baseDirectory: .next
    files:
      - "**/*"
  cache:
    paths:
      - node_modules/**/*

```
